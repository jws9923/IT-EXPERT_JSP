CREATE OR REPLACE PROCEDURE PROC_INPUT_DATA
   ( ID IN VARCHAR2, PASSWORD IN VARCHAR2, NAME IN VARCHAR2,
     JUMIN1 IN CHAR, JUMIN2 IN CHAR)
   IS
   NEWSEQ NUMBER;
 BEGIN
    SELECT SEQ_AUTH.NEXTVAL INTO NEWSEQ
      FROM DUAL;

    INSERT INTO AUTH (SEQ, ID, PASSWORD)
		   VALUES (NEWSEQ, ID, PASSWORD);

	INSERT INTO MEMBER_INFO (FSEQ, NAME, JUMIN1, JUMIN2)
		   VALUES (NEWSEQ, NAME, JUMIN1, JUMIN2);

END;
